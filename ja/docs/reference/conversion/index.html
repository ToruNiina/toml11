        conversion.hpp #

        toml::getやtoml::findでユーザー定義型をサポートするための変換関数を自動定義するマクロを提供します。

        TOML11_DEFINE_CONVERSION_NON_INTRUSIVE(NAME, ...)

        例 #

        namespace foo
{
struct Foo
{
    std::string s;
    double      d;
    int         i;
};
} // foo

TOML11_DEFINE_CONVERSION_NON_INTRUSIVE(foo::Foo, s, d, i)
